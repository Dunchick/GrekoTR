        function translateToGreko(text, fromLang) {
            let result = text;
            let charMap = (fromLang === 'ru') ? charMapRuToGr : charMapEnToGr;
            let digraphs = (fromLang === 'ru') ? ruToGrDigraphs : enToGrDigraphs;

            // 1. –ó–∞–º–µ–Ω–∞ –¥–∏–≥—Ä–∞—Ñ–æ–≤ (–ö–°, –®, SH, TH –∏ —Ç.–¥.)
            digraphs.forEach(pair => { 
                const source = (fromLang === 'ru') ? pair.ru : pair.en;
                result = result.split(source).join(pair.gr); 
            });

            // 2. –ü–æ–±—É–∫–≤–µ–Ω–Ω–∞—è –∑–∞–º–µ–Ω–∞
            let finalStr = '';
            for (let i = 0; i < result.length; i++) {
                const char = result[i];
                finalStr += charMap[char] || char;
            }
            return finalStr;

        function translateGrekoToRu(text) {
            let result = text;
            grToRuDigraphs.forEach(pair => { result = result.split(pair.gr).join(pair.ru); });

                const char = result[i];
                finalStr += charMapGrToRu[char] || char;

            return finalStr;
